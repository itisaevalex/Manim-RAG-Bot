 def construct(self):
 circle = Circle(radius=2)
 line_1 = TangentLine(circle, alpha=0.0, length=4, color=BLUE_D) # right
 line_2 = TangentLine(circle, alpha=0.4, length=4, color=GREEN) # top left
 self.add(circle, line_1, line_2)
