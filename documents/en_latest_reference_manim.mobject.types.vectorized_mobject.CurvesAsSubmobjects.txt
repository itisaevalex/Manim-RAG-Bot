 def construct(self):
 curve = ParametricFunction(lambda t: [t, np.sin(t), 0], t_range=[-PI, PI, 0.01], stroke_width=10)
 new_curve = CurvesAsSubmobjects(curve)
 new_curve.set_color_by_gradient(BLUE, RED)
 self.add(new_curve.shift(UP), curve)
